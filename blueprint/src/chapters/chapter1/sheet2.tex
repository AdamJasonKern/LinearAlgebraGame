\section{MultiplyingByTheZeroVector}

\begin{theorem}
  \label{theorem : smul_zero}
  \lean{LinearAlgebraGame.smul_zero_v}
  \leanok
  \uses{definition : VectorSpace}
  In any vector space V over K, any scalar a multiplied by the zero vector gives the zero vector:$ (a : K) : a \bullet (0 : V) = (0 : V)$
\end{theorem}
